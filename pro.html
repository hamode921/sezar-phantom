async function updateScanner() { try { // إضافة أزرار التبديل داخل القائمة const list = document.getElementById('scanner-list'); if (!document.getElementById('scanner-tabs')) { list.innerHTML = `
`; } const res = await fetch('https://api.binance.com/api/v3/ticker/24hr'); const data = await res.json(); const container = document.getElementById('scanner-items-container'); let htmlContent = ''; if (currentScannerMode === 'meme') { // منطق الميم الأصلي (ارتفاع فوليوم + تذبذب عالي) let candidates = data.filter(c => { const change = parseFloat(c.priceChangePercent); const volume = parseFloat(c.quoteVolume); return c.symbol.endsWith('USDT') && change >= 0 && change <= 15 && volume > 5000000; }).sort((a,b) => b.priceChangePercent - a.priceChangePercent).slice(0, 10); candidates.forEach(c => { htmlContent += createScanItem(c, 'MEME'); }); } else { // --- منطق SPOT EXPLOSION (تجميع هادئ + سيولة حيتان) --- let spotExplosion = data.filter(c => { const change = parseFloat(c.priceChangePercent); const volume = parseFloat(c.quoteVolume); // المعيار: سعر مستقر (-2% إلى +3%) + سيولة ضخمة (> 10M) return c.symbol.endsWith('USDT') && change >= -2 && change <= 3 && volume > 10000000; }).sort((a,b) => b.quoteVolume - a.quoteVolume).slice(0, 10); spotExplosion.forEach(c => { htmlContent += createScanItem(c, 'EXPLOSION'); }); } container.innerHTML = htmlContent; } catch (e) { console.error("Scanner Error:", e); } } // دالة مساعدة لتغيير المود function setScannerMode(mode) { currentScannerMode = mode; document.getElementById('tab-meme').classList.toggle('active', mode === 'meme'); document.getElementById('tab-spot').classList.toggle('active', mode === 'spot'); updateScanner(); } // دالة لبناء عنصر القائمة بشكل احترافي function createScanItem(c, type) { const change = parseFloat(c.priceChangePercent); const isExplosion = type === 'EXPLOSION'; const borderColor = isExplosion ? 'var(--ghost)' : 'var(--green)'; const tagText = isExplosion ? 'ACCUMULATION 🐳' : 'PUMPING 🚀'; return `
${c.symbol} ${tagText}
CHANGE: ${change}% | VOL: ${(parseFloat(c.quoteVolume)/1000000).toFixed(1)}M
`; }
