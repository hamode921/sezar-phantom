// --- تعديل وظيفة الفلتر في قسم الانفجار --- async function updateScanner() { try { const res = await fetch('https://api.binance.com/api/v3/ticker/24hr'); const data = await res.json(); const list = document.getElementById('scanner-list'); let candidates; if (currentScannerTab === 'meme') { // فلتر الميم كما هو candidates = data.filter(c => { const change = parseFloat(c.priceChangePercent); const volume = parseFloat(c.quoteVolume); return c.symbol.endsWith('USDT') && change >= 0 && change <= 15 && volume > 5000000; }).map(c => ({ ...c, type: 'MEME' })).slice(0, 10); } else { // --- فلتر الانفجار المطور لمنع العملات الميتة --- candidates = data.filter(c => { const change = parseFloat(c.priceChangePercent); const volume = parseFloat(c.quoteVolume); const high = parseFloat(c.highPrice); const low = parseFloat(c.lowPrice); const last = parseFloat(c.lastPrice); // شرط منع العملات اللي شارتها خطوط (التذبذب لازم يكون منطقي مو صفر) const priceActivity = (high - low) / last; return c.symbol.endsWith('USDT') && change >= -1 && change <= 2 && volume > 15000000 && // رفعنا السيولة لـ 15 مليون لضمان حركة حقيقية priceActivity > 0.002; // استبعاد العملات اللي ما فيها أي حركة (الخطوط الميتة) }).map(c => ({ ...c, type: 'EXPLOSION' })).sort((a,b) => parseFloat(b.quoteVolume) - parseFloat(a.quoteVolume)).slice(0, 10); } // ... (بقية كود عرض القائمة)
