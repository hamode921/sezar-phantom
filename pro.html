${isHot ? '🛰️ ' : ''}${c.symbol} PRE-PUMP
🔥 صعود هادئ: ${change}%
سيولة ذكية: ${volM}M
`; }); if(htmlContent !== '') list.innerHTML = htmlContent; } catch (e) { console.log("Scanner Error: ", e); } }
