async function updateScanner() { try { const res = await fetch('https://api.binance.com/api/v3/ticker/24hr'); const data = await res.json(); const list = document.getElementById('scanner-list'); let candidates; if (currentScannerTab === 'meme') { candidates = data.filter(c => { const change = parseFloat(c.priceChangePercent); const volume = parseFloat(c.quoteVolume); return c.symbol.endsWith('USDT') && change >= 0 && change <= 15 && volume > 5000000; }).map(c => { const currentVol = parseFloat(c.quoteVolume); const prevVol = lastVolMap.get(c.symbol) || currentVol; const volIncrease = prevVol > 0 ? ((currentVol - prevVol) / prevVol) * 100 : 0; lastVolMap.set(c.symbol, currentVol); return { ...c, volIncrease, type: 'MEME' }; }).sort((a,b) => b.volIncrease - a.volIncrease).slice(0, 10); } else { // --- التعديل هنا: فلتر "القاع المستقر" لضمان عدم الشراء والسعر ينهار --- candidates = data.filter(c => { const change = parseFloat(c.priceChangePercent); const volume = parseFloat(c.quoteVolume); const high = parseFloat(c.highPrice); const low = parseFloat(c.lowPrice); const last = parseFloat(c.lastPrice); // حساب القرب من القاع: هل السعر ضمن آخر 10% من نطاق اليوم؟ const priceRange = high - low; const positionInDayRange = (last - low) / priceRange; return c.symbol.endsWith('USDT') && volume > 20000000 && // سيولة عالية لضمان وجود حيتان change > -1.5 && // استبعاد العملات التي تهبط بقوة (أكثر من 1.5% هبوط يخرجها من القائمة) positionInDayRange < 0.15; // السعر بالقرب من قاعه اليومي }).map(c => ({ ...c, volIncrease: 0, type: 'EXPLOSION' })) .sort((a,b) => parseFloat(b.quoteVolume) - parseFloat(a.quoteVolume)).slice(0, 10); } if (candidates.length > 0) { let htmlContent = ''; candidates.forEach(c => { const change = parseFloat(c.priceChangePercent); const isExplosion = c.type === 'EXPLOSION'; const isSpike = c.volIncrease > 0.8; const isBoldEntry = (isSpike && change > 2) || (isExplosion); const borderColor = isExplosion ? 'var(--ghost)' : 'var(--green)'; const tagLabel = isExplosion ? 'ACCUMULATION 🐳' : 'BOLD ENTRY 🔥'; htmlContent += `
${(isSpike || isExplosion) ? '🚀 ' : ''}${c.symbol} ${isBoldEntry ? `${tagLabel}` : ''}
PRE-PUMP: ${change}%
VOLUME: ${(parseFloat(c.quoteVolume)/1000000).toFixed(2)}M
`; }); list.innerHTML = htmlContent; } } catch (e) {} }
