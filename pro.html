// --- الجزء المطور لقنص العملات الرخيصة وبيانات الحيتان --- let blackList = ['USDC', 'FDUSD', 'TUSD', 'DAI', 'EUR', 'GBP']; // عملات شارتها ميت async function updateScanner() { try { const res = await fetch('https://api.binance.com/api/v3/ticker/24hr'); const data = await res.json(); const list = document.getElementById('scanner-list'); // تعديل الفلتر: عملات اقل من 2 دولار، سيولة قوية، وليست عملة مستقرة let candidates = data.filter(c => { const price = parseFloat(c.lastPrice); const volume = parseFloat(c.quoteVolume); const isStable = blackList.some(stable => c.symbol.includes(stable)); return c.symbol.endsWith('USDT') && price > 0 && price <= 2.0 && volume > 5000000 && !isStable; }).map(c => { const currentVol = parseFloat(c.quoteVolume); const prevVol = lastVolMap.get(c.symbol) || currentVol; const volIncrease = prevVol > 0 ? ((currentVol - prevVol) / prevVol) * 100 : 0; lastVolMap.set(c.symbol, currentVol); return { ...c, volIncrease }; }).sort((a,b) => b.volIncrease - a.volIncrease).slice(0, 12); if (candidates.length > 0) { let htmlContent = ''; candidates.forEach(c => { const change = parseFloat(c.priceChangePercent); const price = parseFloat(c.lastPrice); // وسم "فرصة تدبيل" إذا كانت العملة رخيصة جداً وبها دخول سيولة const doubleChance = (price < 0.5 && c.volIncrease > 0.5); htmlContent += `
${c.symbol} ${doubleChance ? 'فرصة تدبيل 🚀' : ''}
السعر: $${price}
تغير السيولة: +${c.volIncrease.toFixed(2)}%
`; }); list.innerHTML = htmlContent; } } catch (e) { console.error("Scanner Error", e); } } // إصلاح حقل N/A في أوامر الحيتان async function updateLiquidity() { try { const url = `https://api.binance.com/api/v3/depth?symbol=${currentSymbol}&limit=1000`; const res = await fetch(url); const data = await res.json(); let bids = data.bids.map(b => ({p: parseFloat(b[0]), v: parseFloat(b[1])})); let asks = data.asks.map(a => ({p: parseFloat(a[0]), v: parseFloat(a[1])})); // حساب حجم سيولة الحيتان الفعلي (الأوامر التي تزيد عن 50 ألف دولار) let whaleLongs = bids.filter(b => (b.p * b.v) > 50000).length; let whaleShorts = asks.filter(a => (a.p * a.v) > 50000).length; let tB = bids.reduce((s,b)=>s+(b.p*b.v),0); let tS = asks.reduce((s,a)=>s+(a.p*a.v),0); // عرض الأرقام بدلاً من N/A document.getElementById('short-whales').innerText = whaleShorts > 0 ? whaleShorts : "0"; document.getElementById('long-whales').innerText = whaleLongs > 0 ? whaleLongs : "0"; // تحديث باقي الواجهة... let buyP = (tB/(tB+tS)*100); document.getElementById('buy-sell-display').innerText = `BUY: ${buyP.toFixed(1)}% | SELL: ${(100-buyP).toFixed(1)}%`; // منطق الـ AI المحسن للعملات الرخيصة const aiDec = document.getElementById('ai-decision'); if(buyP > 65 && whaleLongs > whaleShorts) { aiDec.innerText = "🚀 قنص: تدبيل محتمل"; aiDec.style.color = "var(--green)"; } else if (buyP < 35) { aiDec.innerText = "⚠️ هروب: تصريف حيتان"; aiDec.style.color = "var(--red)"; } else { aiDec.innerText = "⌛ انتظار: تجميع ممل"; aiDec.style.color = "var(--gold)"; } } catch (e) { console.log("Liquidity Error"); } }
